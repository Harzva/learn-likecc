🔧 /commit 流程

async function createCommit(message?: string) {
  // 1. 获取变更状态
  const status = await getGitStatus()

  // 2. 获取 diff
  const diff = await getGitDiff()

  // 3. 生成提交信息 (如果没有提供)
  if (!message) {
    message = await generateCommitMessage(diff, status)
  }

  // 4. 添加文件并提交
  await exec('git add -A')
  await exec(`git commit -m "${message}"`)
}
