## Compact 服务详解

源码中的 Compact 服务是核心：

```typescript
class CompactService {
  private readonly THRESHOLD = 0.8  // 80% 时触发压缩

  async compactIfNeeded(messages: Message[]): Promise<Message[]> {
    const tokens = this.countTokens(messages)
    const limit = this.getTokenLimit()

    if (tokens > limit * this.THRESHOLD) {
      return this.compact(messages)
    }

    return messages
  }

  private async compact(messages: Message[]): Promise<Message[]> {
    // 1. 识别可压缩的消息
    const { keep, compress } = this.classifyMessages(messages)

    // 2. 压缩旧消息
    const compressed = await this.compressMessages(compress)

    // 3. 组合结果
    return [...compressed, ...keep]
  }
}
```

### 压缩策略

```typescript
interface CompactStrategy {
  // 完全保留
  keep: Message[],

  // 摘要压缩
  summarize: Message[],

  // 完全压缩
  compress: Message[],
}

function classifyMessages(messages: Message[]): CompactStrategy {
  const now = Date.now()

  return {
    // 最近 10 条消息保持完整
    keep: messages.slice(-10),

    // 中间消息生成摘要
    summarize: messages.slice(-50, -10).filter(m =>
      now - m.timestamp < 1000 * 60 * 30  // 30 分钟内
    ),

    // 更早的消息高度压缩
    compress: messages.slice(0, -50),
  }
}
```

async function readFileOptimized(path: string): Promise<string> {

const fileCache = new LRUCache<string, FileContent>({
  max: 100,
  maxSize: 1024 * 1024,  // 1MB
})

async function readFile(path: string): Promise<string> {
  const cached = fileCache.get(path)
  if (cached) return cached

  const content = await fs.readFile(path, 'utf-8')
  fileCache.set(path, content)
  return content
}

## 最佳实践

### 对于用户

1. **及时清理对话**：开始新任务时开启新对话
2. **使用 CLAUDE.md**：把重要信息写在项目配置中
3. **避免重复**：不要让 Claude 重复读取相同文件
